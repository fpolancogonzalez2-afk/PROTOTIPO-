    function ejecutarLogin() {

            usuarioActivo = usuarioEncontrado;

                navegarA('vista-login', 'vista-materias');
            }

        }
    }

    function cargarOfertaAcademica() {
        let datosCarrera = bd_carreras[usuarioActivo.carrera_id];
        document.getElementById('lbl-carrera-actual').innerText = datosCarrera.nombre.toUpperCase();
        document.getElementById('contador-uc').innerText = "0";
        document.getElementById('alerta-materias').style.display = "none";
        
        let contenedorLista = document.getElementById('lista-materias-dinamica');
        contenedorLista.innerHTML = ""; 

        datosCarrera.materias.forEach(materia => {
            let divCard = document.createElement('div');
            divCard.className = 'materia-card';
            
            let chk = document.createElement('input');
            chk.type = 'checkbox';
            chk.className = 'chk-materia';
            chk.value = materia.uc;
            chk.id = 'mat-' + materia.id;
            
            chk.dataset.dia = materia.dia;
            chk.dataset.hora = materia.hora;
            chk.dataset.nombre = materia.nombre;
            
            // MÓDULO A: Verificación algorítmica de cumplimiento de prelación
            let cumplePrelacion = !materia.prelacion || usuarioActivo.aprobadas.includes(materia.prelacion);

            if(!cumplePrelacion) {
                chk.disabled = true;
                divCard.style.opacity = "0.4";
                divCard.style.borderLeftColor = "#ff414d";
            } else {
                chk.onchange = calcularCreditosYHorario;
            }

            let lbl = document.createElement('label');
            lbl.htmlFor = chk.id;
            lbl.innerText = materia.nombre + " (" + materia.uc + " UC)" + (!cumplePrelacion ? " 🔒" : "");
            lbl.style.cursor = cumplePrelacion ? "pointer" : "not-allowed";

            divCard.appendChild(chk);
            divCard.appendChild(lbl);
            contenedorLista.appendChild(divCard);
        });
    }

    function calcularCreditosYHorario() {
        let checkboxes = document.querySelectorAll('.chk-materia:checked');
        let totalUc = 0;
        
        checkboxes.forEach(chk => { totalUc += parseInt(chk.value); });
        document.getElementById('contador-uc').innerText = totalUc;
        
        if(totalUc > LIMITE_UC) {
            mostrarAlerta('alerta-materias', "⚠️ LÍMITE EXCEDIDO: Desmarque materias para no superar las " + LIMITE_UC + " UC.");
        } else {
            document.getElementById('alerta-materias').style.display = "none";
        }
        
        actualizarCuadroHorario();
    }

    function actualizarCuadroHorario() {
        choqueGlobal = false;

        document.querySelectorAll('[id^="celda-"]').forEach(celda => {
            celda.classList.remove('celda-ocupada', 'celda-choque');
            celda.classList.add('celda-libre');
            celda.innerText = '-';
        });

        const seleccionadas = document.querySelectorAll('.chk-materia:checked');
        
        seleccionadas.forEach(chk => {
            const dia = chk.dataset.dia;
            const hora = chk.dataset.hora;
            const nombreCorto = chk.dataset.nombre.substring(0, 5) + "."; 
            
            const celda = document.getElementById(`celda-${dia}-${hora}`);
            if (celda) {
                if (!celda.classList.contains('celda-libre')) {
                    celda.classList.remove('celda-ocupada', 'celda-libre');
                    celda.classList.add('celda-choque');
                    celda.innerText = 'CHOQUE';
                    choqueGlobal = true;
                } else {
                    celda.classList.remove('celda-libre');
                    celda.classList.add('celda-ocupada');
                    celda.innerText = nombreCorto;
                }
            }
        });

        if(choqueGlobal) {
            mostrarAlerta('alerta-materias', "⚠️ ERROR: Tiene materias chocando en el mismo horario (Ver cuadro rojo).");
        } else if (parseInt(document.getElementById('contador-uc').innerText) <= LIMITE_UC) {
            document.getElementById('alerta-materias').style.display = "none";
        }
    }

    function procesarInscripcionFinal() {

        if(uc > LIMITE_UC) { mostrarAlerta('alerta-materias', "Corrija el exceso de UC antes de confirmar."); return; }
        if(choqueGlobal) { mostrarAlerta('alerta-materias', "Resuelva el conflicto de horario detectado."); return; }

        document.getElementById('doc-nombre').innerText = usuarioActivo.nombre.toUpperCase();
        document.getElementById('doc-cedula').innerText = usuarioActivo.cedula;
        document.getElementById('doc-carrera').innerText = bd_carreras[usuarioActivo.carrera_id].nombre.toUpperCase();
        document.getElementById('doc-uc').innerText = uc;

        navegarA('vista-materias', 'vista-exito');
    }

    /* ==========================================================================
       6. TRANSACCIÓN EXTERNA: RENDICIÓN PDF (MÓDULO C)
       ========================================================================= */
    function ejecutarDescargaPDF() {
        const areaCaptura = document.getElementById('comprobante-pdf');
        
        const directricesImpresion = {
            margin: 8,
            filename: `SICE_Comprobante_${usuarioActivo.cedula}.pdf`,
            image: { type: 'jpeg', quality: 0.98 },
            html2canvas: { scale: 2, letterRendering: true, useCORS: true },
            jsPDF: { unit: 'mm', format: 'a5', orientation: 'portrait' }
        };

        html2pdf().set(directricesImpresion).from(areaCaptura).save();
    }

    function cerrarSesion() {
        // MÓDULO B: Destrucción de la cookie sintética local
        localStorage.removeItem('sice_sesion');

        let pantallaActual = "vista-login";
        if(usuarioActivo) {
            pantallaActual = usuarioActivo.rol === "admin" ? "vista-admin" : "vista-materias";
            if(document.getElementById('vista-exito').classList.contains('pantalla-activa')) {
                pantallaActual = "vista-exito";
            }
        }

        usuarioActivo = null;
        document.getElementById('log-cedula').value = ""; 
        document.getElementById('log-clave').value = "";
        
        document.querySelectorAll('.pantalla').forEach(p => p.classList.remove('pantalla-activa'));
        document.getElementById('vista-login').classList.add('pantalla-activa');
    }
